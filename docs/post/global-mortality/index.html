  rowwise() %>%
  dplyr::mutate(hist = 
     numeric.hist %>% 
     utf8ToInt() %>% 
     as.character() %>% 
     paste0("&#", . ,";", collapse = "")) %>%
  dplyr::select(-numeric.hist) %>%
  dplyr::rename_with(~ tolower(gsub("skim_|numeric.|character.", "", .x, fixed = FALSE))) %>%
  mutate_if(is.numeric, round, digits=2) %>%
  dplyr::select(c('type','variable','n_missing','complete_rate','n_unique','hist','mean','sd','min','p25','p50','p75', 'max'))
